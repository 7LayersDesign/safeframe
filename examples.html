Simple Example

 

Here is the rendered ad:

 

How it works

This is the code on the publisher host which defines target div for the SafeFrame ad on this page: 

  <div id="top_leaderboard" style="height:90px;"></div>

This is the JSON object that contains the ad data:

  <script type='text/x-safeframe' class='sf_data' id="sf_tag_1" >
    {
      "id" : "ad1Leader",
      "src" : "http://ad.doubleclick.net/N3790/adj/iab.net;sz=728x90",
      "conf" :
        {
          "size" : "728x90",
          "dest" : "top_leaderboard"
        }
    }
  </script>
Don't forget to configure and initialize SafeFrame at the foot of each page. 
<!-- SafeFrame js include files and script-->

<script type="text/javascript" src="http://sf.wanda.io/js/lib/base.js"></script>
<script type="text/javascript" src="http://sf.wanda.io/js/host/host.js"></script>
<script type="text/javascript" src="http://sf.wanda.io/js/lib/boot.js"></script>

<script type="text/javascript">
  /**
  * Initialize a new SafeFrames publisher configuration
  */
  (function() {
      var conf = new $sf.host.Config({
        // Should be absolute path to render file hosted on CDN
        renderFile: "http://sf.wanda.io/html/r.html",
        positions: {}
      });
  })();
</script>

Expansion Example

 

Here is the rendered ad:

How it works

This is the code on the publisher host which defines target div for the SafeFrame ad on this page: 

  <div id="expandingAdExample" style="height:90px;"></div>

This is the JSON object that contains the ad data:

  <script type='text/x-safeframe' class='sf_data' id="sf_tag_1" >
  {
    "id": "expandingAdFrame",
     // relative source is relative to the r.html bootstrap file
    "src": "../js/lib/expandingAd.js",
    "conf":
      {
        "w": "468",
        "h": "90",
        "dest": "expandingAdExample"
      },
    "meta":
      {
        "expand":
        {
          "x": "350",
          "y": "0"
        }
      }
  }
  </script>
Don't forget to configure and initialize SafeFrame at the foot of each page.

<script type="text/javascript" src="http://sf.wanda.io/js/host/host.js"></script>
<script type="text/javascript" src="http://sf.wanda.io/js/lib/boot.js"></script>

<script type="text/javascript">
  /**
  * Initialize a new SafeFrames publisher configuration
  */
  (function() {
      var conf = new $sf.host.Config({
        // Should be absolute path to render file hosted on CDN
        renderFile: "http://sf.wanda.io/html/r.html",
